### 1. 环境准备（Pre-test Setup）

环境配置由 **conftest.py** 中的 `EnvConfig` / `get_current_env()` 提供（已对齐 .env）。本 agent 不再单独维护 EnvConfig 副本，避免双源漂移。

环境健康检查脚本位置：`workspace/自动化脚本/python/scripts/health_check.sh`

```bash
#!/bin/bash
# 健康检查（独立 .sh 文件，避免 heredoc 嵌套引号问题）
set -euo pipefail

API_URL="${TEST_API_URL:-http://test-api.example.com}"
DB_HOST="${TEST_DB_HOST:-localhost}"
DB_PORT="${TEST_DB_PORT:-5432}"
DB_USER="${TEST_DB_USER:-testuser}"
DB_NAME="${TEST_DB_NAME:-testdb}"

echo "=== 测试环境健康检查 ==="

# 1) 应用可达性
HTTP_CODE=$(curl -s -o /dev/null -w "%{http_code}" --max-time 5 "$API_URL/health" || echo "000")
if [[ "$HTTP_CODE" =~ ^2 ]]; then
    echo "✅ 应用 $API_URL/health (HTTP $HTTP_CODE)"
else
    echo "❌ 应用 $API_URL/health (HTTP $HTTP_CODE)"
    exit 1
fi

# 2) 数据库 TCP 探活
if command -v pg_isready &>/dev/null; then
    pg_isready -h "$DB_HOST" -p "$DB_PORT" -U "$DB_USER" -d "$DB_NAME" -t 5 \
        && echo "✅ 数据库 $DB_HOST:$DB_PORT" \
        || { echo "❌ 数据库不可达"; exit 1; }
else
    (echo > "/dev/tcp/$DB_HOST/$DB_PORT") 2>/dev/null \
        && echo "✅ 数据库 TCP $DB_HOST:$DB_PORT" \
        || { echo "❌ 数据库 TCP 失败"; exit 1; }
fi

# 3) Redis
REDIS_HOST="${TEST_REDIS_HOST:-localhost}"
REDIS_PORT="${TEST_REDIS_PORT:-6379}"
if command -v redis-cli &>/dev/null; then
    redis-cli -h "$REDIS_HOST" -p "$REDIS_PORT" -t 3 ping | grep -q PONG \
        && echo "✅ Redis $REDIS_HOST:$REDIS_PORT" \
        || { echo "❌ Redis 不可达"; exit 1; }
fi

# 4) Mock 服务（可选）
if [[ -n "${MOCK_SERVER_URL:-}" ]]; then
    HTTP_CODE=$(curl -s -o /dev/null -w "%{http_code}" --max-time 3 "$MOCK_SERVER_URL/__admin/health" || echo "000")
    [[ "$HTTP_CODE" =~ ^2 ]] && echo "✅ Mock $MOCK_SERVER_URL" || echo "⚠️ Mock 未启用"
fi

echo "=== 环境检查通过 ==="
```

### 2. Python 健康检查（更精细，落地 JSON 报告）

```python
# 调用 utils/jmeter_csv_exporter 等工具的相同 import 模式（部署后 utils/ 在 sys.path）
import json
import logging
import time
from datetime import datetime
from pathlib import Path

logger = logging.getLogger(__name__)


def health_check(env_config) -> dict:
    """执行环境健康检查，返回报告 dict 并写盘。"""
    import requests
    report = {
        "timestamp": datetime.now().isoformat(),
        "environment": env_config.env_name,
        "status": "ready",
        "checks": {},
    }
    failed = []

    # 应用
    t0 = time.time()
    try:
        r = requests.get(f"{env_config.api_base_url}/health", timeout=5)
        latency_ms = int((time.time() - t0) * 1000)
        ok = r.ok
        report["checks"]["application"] = {"status": "ok" if ok else "fail",
                                            "http_code": r.status_code,
                                            "latency_ms": latency_ms}
        if not ok:
            failed.append("application")
    except Exception as e:
        report["checks"]["application"] = {"status": "fail", "error": str(e)}
        failed.append("application")

    # 数据库
    try:
        from sqlalchemy import create_engine, text
        url = (f"postgresql+psycopg2://{env_config.db_user}:{env_config.db_password}"
               f"@{env_config.db_host}:{env_config.db_port}/{env_config.db_name}")
        engine = create_engine(url, pool_pre_ping=True)
        t0 = time.time()
        with engine.connect() as conn:
            conn.execute(text("SELECT 1"))
        report["checks"]["database"] = {"status": "ok",
                                         "latency_ms": int((time.time() - t0) * 1000)}
    except Exception as e:
        report["checks"]["database"] = {"status": "fail", "error": str(e)}
        failed.append("database")

    # Redis
    try:
        import redis
        r = redis.Redis(host=env_config.redis_host, port=env_config.redis_port, socket_timeout=3)
        t0 = time.time()
        r.ping()
        report["checks"]["redis"] = {"status": "ok",
                                      "latency_ms": int((time.time() - t0) * 1000)}
    except Exception as e:
        report["checks"]["redis"] = {"status": "fail", "error": str(e)}

    # Mock（可选）
    if env_config.mock_server_url:
        try:
            r = requests.get(f"{env_config.mock_server_url}/__admin/health", timeout=3)
            report["checks"]["mock_server"] = {"status": "ok" if r.ok else "fail"}
        except Exception as e:
            report["checks"]["mock_server"] = {"status": "skip", "error": str(e)}

    if failed:
        report["status"] = "fail"
        report["failed_checks"] = failed

    out = Path(f"workspace/执行日志/环境检查_{datetime.now():%Y%m%d_%H%M%S}.json")
    out.parent.mkdir(parents=True, exist_ok=True)
    out.write_text(json.dumps(report, indent=2, ensure_ascii=False), encoding="utf-8")
    logger.info(f"环境健康报告：{out}")
    return report
```

## ⚠️ 异常处理机制

### 环境准备失败处理（带指数退避）

```python
import logging
import time

logger = logging.getLogger(__name__)


def prepare_environment_with_retry(env_config, max_retries: int = 3) -> bool:
    """带指数退避的环境准备：10s/20s/40s（与全栈重试策略对齐）"""
    for attempt in range(max_retries):
        try:
            report = health_check(env_config)
            if report["status"] == "ready":
                logger.info(f"环境准备成功: {env_config.env_name}")
                return True
            logger.warning(f"环境未就绪（尝试 {attempt + 1}/{max_retries}）: {report.get('failed_checks')}")
        except Exception as e:
            logger.warning(f"环境检查异常 (尝试 {attempt + 1}/{max_retries}): {e}")
        if attempt < max_retries - 1:
            wait = min(10 * (2 ** attempt), 60)
            logger.info(f"等待 {wait}s 后重试")
            time.sleep(wait)

    logger.error("环境准备失败，已达最大重试次数，阻止测试执行")
    return False
```

### 异常处理原则

- ✅ 环境不就绪时**阻止**测试执行（避免误报）
- ✅ 重试 3 次，使用指数退避策略 10/20/40s
- ✅ 记录详细的环境状态日志
- ✅ 测试结束后无论成功失败都执行清理
- ❌ 不得跳过环境检查直接执行测试
